# Instruções do Lab 6


# Abra o terminal ou prompt de comando e navegue até a pasta onde você colocou os arquivos do Lab6 (não use espaço ou acento em nome de pasta)


# Execute o comando abaixo para criar a imagem Docker

docker build -t img-modelagem-gcp:lab6 .


# Execute o comando abaixo para criar o container Docker

docker run -dit --name gcp-dw-lab6 -v ./Lab6:/lab6 img-modelagem-gcp:lab6 /bin/bash


# Verifique as versões

terraform version
gcloud version


# Processo para autenticação de usuário (gcloud) dentro do container:

1. Comando para autenticação:
gcloud auth application-default login

2. Definir a cota do projeto no Google Cloud:
gcloud config set project [ID DO SEU PROJETO NO CGP]

gcloud projects list

3. terraform init

4. terraform plan

5. terraform apply
